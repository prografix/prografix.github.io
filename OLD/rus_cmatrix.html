Класс CMatrix

Этот класс аналогичен классу Matrix и предназначен для представления комплексных матриц: 

class CMatrix
{
    Complex ** p;
// Запрет оператора присваивания
    void operator = ( const CMatrix & );
public:
    const unsigned int nRow; // к-во строк
    const unsigned int nCol; // к-во столбцов

    CMatrix ( unsigned int r, unsigned int c );
    CMatrix ( unsigned int r, unsigned int c, const Complex * const * a );
    CMatrix ( const CMatrix & );
   ~CMatrix ();

    void fill ( const Complex & );
    CMatrix & operator *= ( Complex );
    operator       Complex * const * ()       { return p; }
    operator const Complex * const * () const { return p; }
};

// Разные функции

bool copy   ( const CMatrix & a, CMatrix & b ); // b = a
bool trans  ( const CMatrix & a, CMatrix & b ); // b = aT

bool  plus ( const CMatrix & a, const CMatrix & b, CMatrix & c ); // c = a + b;
bool minus ( const CMatrix & a, const CMatrix & b, CMatrix & c ); // c = a - b;
bool multi ( const CMatrix & a, const CMatrix & b, CMatrix & c ); // c = a * b;

Комплексное транспонирование не только переставляет элементы матрицы, но и сопрягает их.

Следующие две функции lss_h решают задачу наименьших квадратов при помощи преобразований Хаусхолдера. Первая из них изменяет исходные данные ( матрицу ), а вторая - нет: 

bool lss_h ( CMatrix & a, const Complex * b, Complex * x ); // b[a.nRow], x[a.nCol]

bool lss_h ( unsigned int r, unsigned int c, const Complex * const * a, const Complex * b, Complex * x );

Эти же функции решают систему линейных уравнений в случае, когда к-во строк матрицы равно к-ву столбцов.
